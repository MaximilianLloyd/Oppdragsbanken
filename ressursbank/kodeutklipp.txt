// Henter bildekarusellelementet fra HTML-dokumentet
const bildekarusell = document.querySelector('.bildekarusell');

// Henter alle miniatyrbildelementene fra HTML-dokumentet
const miniatyrbilder = document.querySelectorAll('.miniatyrbilde');

// Henter bakgrunnsområdet for bildekarusellen fra HTML-dokumentet
const bakgrunn = document.querySelector('.bildekarusell');

const bilder = document.querySelectorAll('.karusellbilde');

// Legger til klikk-lytter på bakgrunnsområdet for bildekarusellen
bakgrunn.addEventListener('click', (event) => {
    // Sjekker om klikket var på bakgrunnsområdet, og ikke på et bilde i bildekarusellen
    if (event.target === bakgrunn) {
        // Skjuler bildekarusellen hvis klikket var utenfor
        bildekarusell.style.display = 'none';
    }
});

// Legger til klikk-lytter på alle miniatyrbildene
miniatyrbilder.forEach((miniatyrbilde) => {
    miniatyrbilde.addEventListener('click', () => {
       // Viser bildekarusellen når miniatyrbildet blir klikket på
       bildekarusell.style.display = 'flex'; 
       // Viser valgte bilde i bildekarusellen
       visBilde(miniatyrbilde.src);
    });
});

// Legger til klikk-lytter på alle bildeelementene i bildekarusellen
bilder.forEach((bilde) => {
    bilde.addEventListener('click', (event) => {
        // Stopper klikk-lytteren på bakgrunnen fra å kjøre når et bilde i bildekarusellen blir klikket på
        event.stopPropagation();
        // Skjuler bildekarusellen når et bilde i bildekarusellen blir klikket på
        bildekarusell.style.display = 'none'; 
    });
});

// Funksjon som viser bilde i bildekarusellen
function visBilde(bildeSrc) {
    const visningsBilde = bildekarusell.querySelector('img');
    visningsBilde.src = bildeSrc;
}
