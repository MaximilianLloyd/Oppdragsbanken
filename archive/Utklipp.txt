Hei jeg jobber med en interaktiv timeplan. Akkurat nå prøver jeg å implementere en mobilvisning, slik at kun en dag vises om gangen. Tanker en at så fort den visuelle bredden av nettsiden synker under 1100px, så gjøres dagene som ikke er i fokus usynlig. I tilegg vil navigasjonspiler (.pilNeste, .pilForrige) bli synlig (de vises ikke utenfor mobilformat), slik at brukeren kan navigere mellom dagene ved å klikke på knappen. Dersom brukeren er på siste dagen, altså fredag, vil de loope tilbake til mandag. Jeg har begynt å skrive logikken, men kan du være så snill å hjelpe meg med å fullføre den?

JavaScript:
const timeplanDager = document.querySelectorAll(".dag");
const antallDager = timeplanDager.length;
const dagerPerSide = 1;

let gjeldendeDag = 0;
let gjeldendeSide = 1;

for (let i = 0; i < antallDager; i++) {
   if (i < 1) {
      timeplanDager[i].style.display = "block";
   } else {
      timeplanDager[i].style.display = "none";
   }
}

const nesteDag = document.querySelector(".pilNeste");
const forrigeDag = document.querySelector(".pilForrige");

nesteDag.addEventListener("click", function () {
   if (gjeldendeDag + dagerPerSide < antallDager) {
      gjeldendeDag += dagerPerSide;
      gjeldendeSide++;
   } else {
      gjeldendeDag = 0;
      gjeldendeSide = 1;
   }
   oppdaterDager();
});

forrigeDag.addEventListener("click", function () {
   if (gjeldendeDag - dagerPerSide >= 0) {
      gjeldendeDag -= dagerPerSide;
      gjeldendeSide--;
   } else {
      gjeldendeDag = antallDager - dagerPerSide;
      gjeldendeSide = antallSider;
   }
   oppdaterDager();
});

function oppdaterDager() {
   for (let i = 0; i < antallDager; i++) {
      if (i >= gjeldendeDag && i < gjeldendeDag + dagerPerSide) {
         timeplanDager[i].style.display = "block";
      } else {
         timeplanDager[i].style.display = "none";
      }
   }
   if (gjeldendeSide === 1) {
      forrigeDag.disabled = true;
   } else {
      forrigeDag.disabled = false;
   }
   if (gjeldendeSide === antallSider) {
      nesteDag.disabled = true;
   } else {
      nesteDag.disabled = false;
   }
}
